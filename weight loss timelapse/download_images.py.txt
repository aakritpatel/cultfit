# Directory to save the downloaded images
save_dir = "downloaded_images"
os.makedirs(save_dir, exist_ok=True)

for url in urls:
    try:
        response = requests.get(url, stream=True)
        response.raise_for_status()  # Check for HTTP request errors

        # Extract the class ID from the URL
        class_id = None
        if "paid-" in url:
            start = url.find("paid-") + len("paid-")
            class_id = url[start:].split('/')[0]

        if not class_id:
            raise ValueError("Class ID could not be determined from the URL.")

        # Save the file with the class ID as the filename and appropriate extension
        extension = os.path.splitext(url.split('/')[-1])[-1]  # Get the file extension
        filename = os.path.join(save_dir, f"{class_id}{extension}")

        with open(filename, "wb") as file:
            for chunk in response.iter_content(chunk_size=8192):  # Download in chunks
                file.write(chunk)

        print(f"Downloaded: {filename}")
    except Exception as e:
        print(f"Failed to download {url}: {e}")
